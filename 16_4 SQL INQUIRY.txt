SELECT IL.NAME, I.SUMMARY
FROM ISSUELISTS IL
JOIN ISSUE I ON IL.ID = I.ISSUELIST_ID;

SELECT I.SUMMARY, U.ID
FROM ISSUE I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUE_NUMBER 
FROM ISSUE I JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY USER_ID_ASSIGNEDTO;
